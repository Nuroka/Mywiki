99private:

146 void reset()
147 {
148 m_init_done = false;
149 m_term_done = false;
150 m_start_count = 0;
151 m_done_count = 0;
152 m_task_count = 0;
153 m_callback = nullptr;
154 m_callback_last_value = 0.0f;
155 m_callback_min_diff = 1.0f;
156 }
157
167 void init(std::function<unsigned int(void)> init_func)
168 {
169 std::lock_guard<std::mutex> lck(m_lock);
170 if (!m_init_done)
171 {
172 m_task_count = init_func();
173 m_init_done = true;
174 }
175 }
176
186 void init(unsigned int task_count, astcenc_progress_callback callback)
187 {
188 std::lock_guard<std::mutex> lck(m_lock);
189 if (!m_init_done)
190 {
191 m_callback = callback;
192 m_task_count = task_count;
193 m_init_done = true;
194
195 // Report every 1% or 4096 blocks, whichever is larger, to avoid callback overhead
196 float min_diff = (4096.0f / static_cast<float>(task_count)) * 100.0f;
197 m_callback_min_diff = astc::max(min_diff, 1.0f);
198 }
199 }
200
211 unsigned int get_task_assignment(unsigned int granule, unsigned int& count)
212 {
213 unsigned int base = m_start_count.fetch_add(granule, std::memory_order_relaxed);
214 if (base >= m_task_count)
215 {
216 count = 0;
217 return 0;
218 }
219
220 count = astc::min(m_task_count - base, granule);
221 return base;
222 }
223
232 void complete_task_assignment(unsigned int count)
233 {
234 // Note: m_done_count cannot use an atomic without the mutex; this has a race between the
235 // update here and the wait() for other threads
236 unsigned int local_count;
237 float local_last_value;
238 {
239 std::unique_lock<std::mutex> lck(m_lock);
240 m_done_count += count;
241 local_count = m_done_count;
242 local_last_value = m_callback_last_value;
243
244 if (m_done_count == m_task_count)
245 {
246 // Ensure the progress bar hits 100%
247 if (m_callback)
248 {
249 std::unique_lock<std::mutex> cblck(m_callback_lock);
250 m_callback(100.0f);
251 m_callback_last_value = 100.0f;
252 }
253
254 lck.unlock();
255 m_complete.notify_all();
256 }
257 }
258
259 // Process progress callback if we have one
260 if (m_callback)
261 {
262 // Initial lockless test - have we progressed enough to emit?
263 float num = static_cast<float>(local_count);
264 float den = static_cast<float>(m_task_count);
265 float this_value = (num / den) * 100.0f;
266 bool report_test = (this_value - local_last_value) > m_callback_min_diff;
267
268 // Recheck under lock, because another thread might report first
269 if (report_test)
270 {
271 std::unique_lock<std::mutex> cblck(m_callback_lock);
272 bool report_retest = (this_value - m_callback_last_value) > m_callback_min_diff;
273 if (report_retest)
274 {
275 m_callback(this_value);
276 m_callback_last_value = this_value;
277 }
278 }
279 }
280 }
281
285 void wait()
286 {
287 std::unique_lock<std::mutex> lck(m_lock);
288 m_complete.wait(lck, [this]{ return m_done_count == m_task_count; });
289 }
290
300 void term(std::function<void(void)> term_func)
301 {
302 std::lock_guard<std::mutex> lck(m_lock);
303 if (!m_term_done)
304 {
305 term_func();
306 m_term_done = true;
307 }
308 }
309};
