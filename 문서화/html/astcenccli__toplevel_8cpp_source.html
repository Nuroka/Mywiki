168extern "C" void progress_emitter(
169 float value
170) {
171 const unsigned int bar_size = 25;
172 unsigned int parts = static_cast<int>(value / 4.0f);
173
174 char buffer[bar_size + 3];
175 buffer[0] = '[';
176
177 for (unsigned int i = 0; i < parts; i++)
178 {
179 buffer[i + 1] = '=';
180 }
181
182 for (unsigned int i = parts; i < bar_size; i++)
183 {
184 buffer[i + 1] = ' ';
185 }
186
187 buffer[bar_size + 1] = ']';
188 buffer[bar_size + 2] = '\0';
189
190 printf(" Progress: %s %03.1f%%\r", buffer, static_cast<double>(value));
191 fflush(stdout);
192}

228static void compression_workload_runner(
229 int thread_count,
230 int thread_id,
231 void* payload
232) {
233 (void)thread_count;
234
235 compression_workload* work = static_cast<compression_workload*>(payload);
236 astcenc_error error = astcenc_compress_image(
237 work->context, work->image, &work->swizzle,
238 work->data_out, work->data_len, thread_id);

240 // This is a racy update, so which error gets returned is a random, but it

242 if (error != ASTCENC_SUCCESS)

244 work->error = error;

286static std::string get_slice_filename(
287 const std::string& basename,
288 unsigned int index,
289 bool& error
290) {
291 size_t sep = basename.find_last_of('.');
292 if (sep == std::string::npos)
293 {
294 error = true;
295 return "";
296 }
297
298 std::string base = basename.substr(0, sep);
299 std::string ext = basename.substr(sep);
300 std::string name = base + "_" + std::to_string(index) + ext;
301 error = false;
302 return name;
303}

316static astcenc_image* load_uncomp_file(
317 const char* filename,
318 unsigned int dim_z,
319 bool y_flip,
320 bool& is_hdr,
321 unsigned int& component_count
322) {
323 astcenc_image *image = nullptr;
324
325 // For a 2D image just load the image directly
326 if (dim_z == 1)
327 {
328 image = load_ncimage(filename, y_flip, is_hdr, component_count);
329 }
330 else
331 {
332 bool slice_is_hdr;
333 unsigned int slice_component_count;
334 astcenc_image* slice = nullptr;
335 std::vector<astcenc_image*> slices;
336
337 // For a 3D image load an array of slices
338 for (unsigned int image_index = 0; image_index < dim_z; image_index++)
339 {
340 bool error;
341 std::string slice_name = get_slice_filename(filename, image_index, error);
342 if (error)
343 {
344 print_error("ERROR: Image pattern does not contain file extension: %s\n", filename);
345 break;
346 }
347
348 slice = load_ncimage(slice_name.c_str(), y_flip,
349 slice_is_hdr, slice_component_count);
350 if (!slice)
351 {
352 break;
353 }
354
355 slices.push_back(slice);
356
357 // Check it is not a 3D image
358 if (slice->dim_z != 1)
359 {
360 print_error("ERROR: Image arrays do not support 3D sources: %s\n", slice_name.c_str());
361 break;
362 }
363
364 // Check slices are consistent with each other
365 if (image_index != 0)
366 {
367 if ((is_hdr != slice_is_hdr) || (component_count != slice_component_count))
368 {
369 print_error("ERROR: Image array[0] and [%d] are different formats\n", image_index);
370 break;
371 }
372
373 if ((slices[0]->dim_x != slice->dim_x) ||
374 (slices[0]->dim_y != slice->dim_y) ||
375 (slices[0]->dim_z != slice->dim_z))
376 {
377 print_error("ERROR: Image array[0] and [%d] are different dimensions\n", image_index);
378 break;
379 }
380 }
381 else
382 {
383 is_hdr = slice_is_hdr;
384 component_count = slice_component_count;
385 }
386 }
387
388 // If all slices loaded correctly then repack them into a single image
389 if (slices.size() == dim_z)
390 {
391 unsigned int dim_x = slices[0]->dim_x;
392 unsigned int dim_y = slices[0]->dim_y;
393 int bitness = is_hdr ? 16 : 8;
394 int slice_size = dim_x * dim_y;
395
396 image = alloc_image(bitness, dim_x, dim_y, dim_z);
397
398 // Combine 2D source images into one 3D image
399 for (unsigned int z = 0; z < dim_z; z++)
400 {
401 if (image->data_type == ASTCENC_TYPE_U8)
402 {
403 uint8_t* data8 = static_cast<uint8_t*>(image->data[z]);
404 uint8_t* data8src = static_cast<uint8_t*>(slices[z]->data[0]);
405 size_t copy_size = slice_size * 4 * sizeof(uint8_t);
406 memcpy(data8, data8src, copy_size);
407 }
408 else if (image->data_type == ASTCENC_TYPE_F16)
409 {
410 uint16_t* data16 = static_cast<uint16_t*>(image->data[z]);
411 uint16_t* data16src = static_cast<uint16_t*>(slices[z]->data[0]);
412 size_t copy_size = slice_size * 4 * sizeof(uint16_t);
413 memcpy(data16, data16src, copy_size);
414 }
415 else // if (image->data_type == ASTCENC_TYPE_F32)
416 {
417 assert(image->data_type == ASTCENC_TYPE_F32);
418 float* data32 = static_cast<float*>(image->data[z]);
419 float* data32src = static_cast<float*>(slices[z]->data[0]);
420 size_t copy_size = slice_size * 4 * sizeof(float);
421 memcpy(data32, data32src, copy_size);
422 }
423 }
424 }
425
426 for (auto &i : slices)
427 {
428 free_image(i);
429 }
430 }
431
432 return image;
433}

445static int parse_commandline_options(
446 int argc,
447 char **argv,
448 astcenc_operation& operation,
449 astcenc_profile& profile
450) {
451 assert(argc >= 2); (void)argc;
452
453 profile = ASTCENC_PRF_LDR;
454 operation = ASTCENC_OP_UNKNOWN;
455
456 int modes_count = sizeof(modes) / sizeof(modes[0]);
457 for (int i = 0; i < modes_count; i++)
458 {
459 if (!strcmp(modes[i].opt, argv[1]))
460 {
461 operation = modes[i].operation;
462 profile = modes[i].decode_mode;
463 break;
464 }
465 }
466
467 if (operation == ASTCENC_OP_UNKNOWN)
468 {
469 print_error("ERROR: Unrecognized operation '%s'\n", argv[1]);
470 return 1;
471 }
472
473 return 0;
474}

489static int init_astcenc_config(
490 int argc,
491 char **argv,
492 astcenc_profile profile,
493 astcenc_operation operation,
494 astc_compressed_image& comp_image,
495 astcenc_preprocess& preprocess,
496 astcenc_config& config
497) {
498 unsigned int block_x = 0;
499 unsigned int block_y = 0;
500 unsigned int block_z = 1;
501
502 // For decode the block size is set by the incoming image.
503 if (operation == ASTCENC_OP_DECOMPRESS)
504 {
505 block_x = comp_image.block_x;
506 block_y = comp_image.block_y;
507 block_z = comp_image.block_z;
508 }
509
510 float quality = 0.0f;
511 preprocess = ASTCENC_PP_NONE;
512
513 // parse the command line's encoding options.
514 int argidx = 4;
515 if (operation & ASTCENC_STAGE_COMPRESS)
516 {
517 // Read and decode block size
518 if (argc < 5)
519 {
520 print_error("ERROR: Block size must be specified\n");
521 return 1;
522 }
523
524 int cnt2D, cnt3D;
525 int dimensions = sscanf(argv[4], "%ux%u%nx%u%n",
526 &block_x, &block_y, &cnt2D, &block_z, &cnt3D);
527 // Character after the last match should be a NUL
528 if (!(((dimensions == 2) && !argv[4][cnt2D]) || ((dimensions == 3) && !argv[4][cnt3D])))
529 {
530 print_error("ERROR: Block size '%s' is invalid\n", argv[4]);
531 return 1;
532 }
533
534 // Read and decode search quality
535 if (argc < 6)
536 {
537 print_error("ERROR: Search quality level must be specified\n");
538 return 1;
539 }
540
541 if (!strcmp(argv[5], "-fastest"))
542 {
543 quality = ASTCENC_PRE_FASTEST;
544 }
545 else if (!strcmp(argv[5], "-fast"))
546 {
547 quality = ASTCENC_PRE_FAST;
548 }
549 else if (!strcmp(argv[5], "-medium"))
550 {
551 quality = ASTCENC_PRE_MEDIUM;
552 }
553 else if (!strcmp(argv[5], "-thorough"))
554 {
555 quality = ASTCENC_PRE_THOROUGH;
556 }
557 else if (!strcmp(argv[5], "-verythorough"))
558 {
559 quality = ASTCENC_PRE_VERYTHOROUGH;
560 }
561 else if (!strcmp(argv[5], "-exhaustive"))
562 {
563 quality = ASTCENC_PRE_EXHAUSTIVE;
564 }
565 else if (is_float(argv[5]))
566 {
567 quality = static_cast<float>(atof(argv[5]));
568 }
569 else
570 {
571 print_error("ERROR: Search quality/preset '%s' is invalid\n", argv[5]);
572 return 1;
573 }
574
575 argidx = 6;
576 }
577
578 unsigned int flags = 0;
579
580 // Gather the flags that we need
581 while (argidx < argc)
582 {
583 if (!strcmp(argv[argidx], "-a"))
584 {
585 // Skip over the data value for now
586 argidx++;
587 flags |= ASTCENC_FLG_USE_ALPHA_WEIGHT;
588 }
589 else if (!strcmp(argv[argidx], "-normal"))
590 {
591 flags |= ASTCENC_FLG_MAP_NORMAL;
592 }
593 else if (!strcmp(argv[argidx], "-decode_unorm8"))
594 {
595 flags |= ASTCENC_FLG_USE_DECODE_UNORM8;
596 }
597 else if (!strcmp(argv[argidx], "-rgbm"))
598 {
599 // Skip over the data value for now
600 argidx++;
601 flags |= ASTCENC_FLG_MAP_RGBM;
602 }
603 else if (!strcmp(argv[argidx], "-perceptual"))
604 {
605 flags |= ASTCENC_FLG_USE_PERCEPTUAL;
606 }
607 else if (!strcmp(argv[argidx], "-pp-normalize"))
608 {
609 if (preprocess != ASTCENC_PP_NONE)
610 {
611 print_error("ERROR: Only a single image preprocess can be used\n");
612 return 1;
613 }
614 preprocess = ASTCENC_PP_NORMALIZE;
615 }
616 else if (!strcmp(argv[argidx], "-pp-premultiply"))
617 {
618 if (preprocess != ASTCENC_PP_NONE)
619 {
620 print_error("ERROR: Only a single image preprocess can be used\n");
621 return 1;
622 }
623 preprocess = ASTCENC_PP_PREMULTIPLY;
624 }
625 argidx ++;
626 }
627
628#if defined(ASTCENC_DECOMPRESS_ONLY)
629 flags |= ASTCENC_FLG_DECOMPRESS_ONLY;
630#else
631 // Decompression can skip some memory allocation, but need full tables
632 if (operation == ASTCENC_OP_DECOMPRESS)
633 {
634 flags |= ASTCENC_FLG_DECOMPRESS_ONLY;
635 }
636 // Compression and test passes can skip some decimation initialization
637 // as we know we are decompressing images that were compressed using the
638 // same settings and heuristics ...
639 else
640 {
641 flags |= ASTCENC_FLG_SELF_DECOMPRESS_ONLY;
642 }
643#endif
644
645 astcenc_error status = astcenc_config_init(profile, block_x, block_y, block_z,
646 quality, flags, &config);
647 if (status == ASTCENC_ERR_BAD_BLOCK_SIZE)
648 {
649 print_error("ERROR: Block size '%s' is invalid\n", argv[4]);
650 return 1;
651 }
652 else if (status == ASTCENC_ERR_BAD_DECODE_MODE)
653 {
654 print_error("ERROR: Decode_unorm8 is not supported by HDR profiles\n", argv[4]);
655 return 1;
656 }
657 else if (status == ASTCENC_ERR_BAD_CPU_FLOAT)
658 {
659 print_error("ERROR: astcenc must not be compiled with -ffast-math\n");
660 return 1;
661 }
662 else if (status != ASTCENC_SUCCESS)
663 {
664 print_error("ERROR: Init config failed with %s\n", astcenc_get_error_string(status));
665 return 1;
666 }
667
668 return 0;
669}

682static int edit_astcenc_config(
683 int argc,
684 char **argv,
685 const astcenc_operation operation,
686 cli_config_options& cli_config,
687 astcenc_config& config
688) {
689
690 int argidx = (operation & ASTCENC_STAGE_COMPRESS) ? 6 : 4;
691
692 while (argidx < argc)
693 {
694 if (!strcmp(argv[argidx], "-silent"))
695 {
696 argidx++;
697 cli_config.silentmode = 1;
698 }
699 else if (!strcmp(argv[argidx], "-cw"))
700 {
701 argidx += 5;
702 if (argidx > argc)
703 {
704 print_error("ERROR: -cw switch with less than 4 arguments\n");
705 return 1;
706 }
707
708 config.cw_r_weight = static_cast<float>(atof(argv[argidx - 4]));
709 config.cw_g_weight = static_cast<float>(atof(argv[argidx - 3]));
710 config.cw_b_weight = static_cast<float>(atof(argv[argidx - 2]));
711 config.cw_a_weight = static_cast<float>(atof(argv[argidx - 1]));
712 }
713 else if (!strcmp(argv[argidx], "-a"))
714 {
715 argidx += 2;
716 if (argidx > argc)
717 {
718 print_error("ERROR: -a switch with no argument\n");
719 return 1;
720 }
721
722 config.a_scale_radius = atoi(argv[argidx - 1]);
723 }
724 else if (!strcmp(argv[argidx], "-esw"))
725 {
726 argidx += 2;
727 if (argidx > argc)
728 {
729 print_error("ERROR: -esw switch with no argument\n");
730 return 1;
731 }
732
733 if (strlen(argv[argidx - 1]) != 4)
734 {
735 print_error("ERROR: -esw pattern does not contain 4 characters\n");
736 return 1;
737 }
738
739 astcenc_swz swizzle_components[4];
740 for (int i = 0; i < 4; i++)
741 {
742 switch (argv[argidx - 1][i])
743 {
744 case 'r':
745 swizzle_components[i] = ASTCENC_SWZ_R;
746 break;
747 case 'g':
748 swizzle_components[i] = ASTCENC_SWZ_G;
749 break;
750 case 'b':
751 swizzle_components[i] = ASTCENC_SWZ_B;
752 break;
753 case 'a':
754 swizzle_components[i] = ASTCENC_SWZ_A;
755 break;
756 case '0':
757 swizzle_components[i] = ASTCENC_SWZ_0;
758 break;
759 case '1':
760 swizzle_components[i] = ASTCENC_SWZ_1;
761 break;
762 default:
763 print_error("ERROR: -esw component '%c' is not valid\n", argv[argidx - 1][i]);
764 return 1;
765 }
766 }
767
768 cli_config.swz_encode.r = swizzle_components[0];
769 cli_config.swz_encode.g = swizzle_components[1];
770 cli_config.swz_encode.b = swizzle_components[2];
771 cli_config.swz_encode.a = swizzle_components[3];
772 }
773 else if (!strcmp(argv[argidx], "-ssw"))
774 {
775 argidx += 2;
776 if (argidx > argc)
777 {
778 print_error("ERROR: -ssw switch with no argument\n");
779 return 1;
780 }
781
782 size_t char_count = strlen(argv[argidx - 1]);
783 if (char_count == 0)
784 {
785 print_error("ERROR: -ssw pattern contains no characters\n");
786 return 1;
787 }
788
789 if (char_count > 4)
790 {
791 print_error("ERROR: -ssw pattern contains more than 4 characters\n");
792 return 1;
793 }
794
795 bool found_r = false;
796 bool found_g = false;
797 bool found_b = false;
798 bool found_a = false;
799
800 for (size_t i = 0; i < char_count; i++)
801 {
802 switch (argv[argidx - 1][i])
803 {
804 case 'r':
805 found_r = true;
806 break;
807 case 'g':
808 found_g = true;
809 break;
810 case 'b':
811 found_b = true;
812 break;
813 case 'a':
814 found_a = true;
815 break;
816 default:
817 print_error("ERROR: -ssw component '%c' is not valid\n", argv[argidx - 1][i]);
818 return 1;
819 }
820 }
821
822 config.cw_r_weight = found_r ? 1.0f : 0.0f;
823 config.cw_g_weight = found_g ? 1.0f : 0.0f;
824 config.cw_b_weight = found_b ? 1.0f : 0.0f;
825 config.cw_a_weight = found_a ? 1.0f : 0.0f;
826 }
827 else if (!strcmp(argv[argidx], "-dsw"))
828 {
829 argidx += 2;
830 if (argidx > argc)
831 {
832 print_error("ERROR: -dsw switch with no argument\n");
833 return 1;
834 }
835
836 if (strlen(argv[argidx - 1]) != 4)
837 {
838 print_error("ERROR: -dsw switch does not contain 4 characters\n");
839 return 1;
840 }
841
842 astcenc_swz swizzle_components[4];
843 for (int i = 0; i < 4; i++)
844 {
845 switch (argv[argidx - 1][i])
846 {
847 case 'r':
848 swizzle_components[i] = ASTCENC_SWZ_R;
849 break;
850 case 'g':
851 swizzle_components[i] = ASTCENC_SWZ_G;
852 break;
853 case 'b':
854 swizzle_components[i] = ASTCENC_SWZ_B;
855 break;
856 case 'a':
857 swizzle_components[i] = ASTCENC_SWZ_A;
858 break;
859 case '0':
860 swizzle_components[i] = ASTCENC_SWZ_0;
861 break;
862 case '1':
863 swizzle_components[i] = ASTCENC_SWZ_1;
864 break;
865 case 'z':
866 swizzle_components[i] = ASTCENC_SWZ_Z;

1317 );

1400static void image_preprocess_normalize(
1401 const astcenc_image& input,
1402 astcenc_image& output
1403) {
1404 for (unsigned int z = 0; z < input.dim_z; z++)
1405 {
1406 for (unsigned int y = 0; y < input.dim_y; y++)
1407 {
1408 for (unsigned int x = 0; x < input.dim_x; x++)
1409 {
1410 vfloat4 pixel = image_get_pixel(input, x, y, z);
1411
1412 // Stash alpha component and zero
1413 float a = pixel.lane<3>();
1414 pixel.set_lane<3>(0.0f);
1415
1416 // Decode [0,1] normals to [-1,1]
1417 pixel.set_lane<0>((pixel.lane<0>() * 2.0f) - 1.0f);
1418 pixel.set_lane<1>((pixel.lane<1>() * 2.0f) - 1.0f);
1419 pixel.set_lane<2>((pixel.lane<2>() * 2.0f) - 1.0f);
1420
1421 // Normalize pixel and restore alpha
1422 pixel = normalize(pixel);
1423 pixel.set_lane<3>(a);
1424
1425 // Encode [-1,1] normals to [0,1]
1426 pixel.set_lane<0>((pixel.lane<0>() + 1.0f) / 2.0f);
1427 pixel.set_lane<1>((pixel.lane<1>() + 1.0f) / 2.0f);
1428 pixel.set_lane<2>((pixel.lane<2>() + 1.0f) / 2.0f);
1429
1430 image_set_pixel(output, x, y, z, pixel);
1431 }
1432 }
1433 }

1442 float a
1443) {
1444 if (a <= 0.04045f)
1445 {
1446 return a * (1.0f / 12.92f);
1447 }

1458 float a
1459) {

1461 {
1462 return a * 12.92f;
1463 }
1464

1478static void image_preprocess_premultiply(
1479 const astcenc_image& input,
1480 astcenc_image& output,
1481 astcenc_profile profile
1482) {
1483 for (unsigned int z = 0; z < input.dim_z; z++)
1484 {
1485 for (unsigned int y = 0; y < input.dim_y; y++)
1486 {
1487 for (unsigned int x = 0; x < input.dim_x; x++)
1488 {
1489 vfloat4 pixel = image_get_pixel(input, x, y, z);
1490
1491 // Linearize sRGB
1492 if (profile == ASTCENC_PRF_LDR_SRGB)
1493 {
1494 pixel.set_lane<0>(srgb_to_linear(pixel.lane<0>()));
1495 pixel.set_lane<1>(srgb_to_linear(pixel.lane<1>()));
1496 pixel.set_lane<2>(srgb_to_linear(pixel.lane<2>()));
1497 }
1498
1499 // Premultiply pixel in linear-space
1500 pixel.set_lane<0>(pixel.lane<0>() * pixel.lane<3>());
1501 pixel.set_lane<1>(pixel.lane<1>() * pixel.lane<3>());
1502 pixel.set_lane<2>(pixel.lane<2>() * pixel.lane<3>());
1503
1504 // Gamma-encode sRGB
1505 if (profile == ASTCENC_PRF_LDR_SRGB)
1506 {
1507 pixel.set_lane<0>(linear_to_srgb(pixel.lane<0>()));
1508 pixel.set_lane<1>(linear_to_srgb(pixel.lane<1>()));
1509 pixel.set_lane<2>(linear_to_srgb(pixel.lane<2>()));
1510 }
1511
1512 image_set_pixel(output, x, y, z, pixel);
1513 }
1514 }
1515 }
